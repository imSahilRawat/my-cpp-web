Question 1: Define a class Complex to represent complex numbers. Overload the + operator to add two complex numbers.

#include <iostream>
using namespace std;

class Complex {
private:
    int real, imag;
public:
    Complex(int r = 0, int i = 0) : real(r), imag(i) {}

    Complex operator+(Complex c) {
        return Complex(real + c.real, imag + c.imag);
    }

    void display() {
        cout << real << " + " << imag << "i" << endl;
    }
};

int main() {
    Complex c1(3, 4);
    Complex c2(2, 5);
    Complex c3;

    cout << "Complex 1: ";
    c1.display();
    cout << "Complex 2: ";
    c2.display();

    c3 = c1 + c2;

    cout << "Sum: ";
    c3.display();

    return 0;
}

Question 2: Create a class Fraction to represent fractions. Overload the + operator to add two fractions.

#include <iostream>
using namespace std;

class Fraction {
private:
    int numerator, denominator;

    int gcd(int a, int b) {
        return (b == 0) ? a : gcd(b, a % b);
    }

    void simplify() {
        int common = gcd(numerator, denominator);
        numerator /= common;
        denominator /= common;
    }

public:
    Fraction(int n = 0, int d = 1) : numerator(n), denominator(d) {
        simplify();
    }

    Fraction operator+(Fraction f) {
        int newNum = numerator * f.denominator + f.numerator * denominator;
        int newDen = denominator * f.denominator;
        return Fraction(newNum, newDen);
    }

    void display() {
        cout << numerator << "/" << denominator << endl;
    }
};

int main() {
    Fraction f1(1, 4);
    Fraction f2(2, 3);
    Fraction f3;

    cout << "Fraction 1: ";
    f1.display();
    cout << "Fraction 2: ";
    f2.display();

    f3 = f1 + f2;

    cout << "Sum: ";
    f3.display();

    return 0;
}

Question 3: Define a class Matrix to represent matrices. Overload the * operator to multiply two matrices.

#include <iostream>
using namespace std;

class Matrix {
private:
    int mat[3][3];
    int rows, cols;

public:
    Matrix(int r = 3, int c = 3) : rows(r), cols(c) {
        for (int i = 0; i < rows; i++)
            for (int j = 0; j < cols; j++)
                mat[i][j] = 0;
    }

    void input() {
        cout << "Enter matrix elements:" << endl;
        for (int i = 0; i < rows; i++)
            for (int j = 0; j < cols; j++)
                cin >> mat[i][j];
    }

    Matrix operator*(Matrix m) {
        Matrix result(rows, m.cols);
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < m.cols; j++) {
                for (int k = 0; k < cols; k++) {
                    result.mat[i][j] += mat[i][k] * m.mat[k][j];
                }
            }
        }
        return result;
    }

    void display() {
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++)
                cout << mat[i][j] << " ";
            cout << endl;
        }
    }
};

int main() {
    Matrix m1, m2, m3;

    cout << "Enter first matrix (3x3):" << endl;
    m1.input();

    cout << "Enter second matrix (3x3):" << endl;
    m2.input();

    m3 = m1 * m2;

    cout << "Product of matrices:" << endl;
    m3.display();

    return 0;
}

Question 4: Implement a class String to represent strings. Overload the + operator to concatenate two strings.

Question 5: Define a class Date to represent dates. Overload the < operator to compare two dates.

#include <iostream>
using namespace std;

class Date {
private:
    int day, month, year;

public:
    Date(int d = 1, int m = 1, int y = 2000) : day(d), month(m), year(y) {}

    bool operator<(Date d) {
        if (year < d.year)
            return true;
        else if (year > d.year)
            return false;

        if (month < d.month)
            return true;
        else if (month > d.month)
            return false;

        return day < d.day;
    }

    void display() {
        cout << day << "/" << month << "/" << year << endl;
    }
};

int main() {
    Date d1(15, 3, 2024);
    Date d2(20, 5, 2024);

    cout << "Date 1: ";
    d1.display();
    cout << "Date 2: ";
    d2.display();

    if (d1 < d2)
        cout << "Date 1 is earlier than Date 2" << endl;
    else
        cout << "Date 1 is not earlier than Date 2" << endl;

    return 0;
}

Question 6: Implement a class Money to represent money. Overload the + operator to add two amounts of money.

#include <iostream>
using namespace std;

class Money {
private:
    int rupees, paise;

public:
    Money(int r = 0, int p = 0) : rupees(r), paise(p) {
        if (paise >= 100) {
            rupees += paise / 100;
            paise = paise % 100;
        }
    }

    Money operator+(Money m) {
        int totalRupees = rupees + m.rupees;
        int totalPaise = paise + m.paise;
        return Money(totalRupees, totalPaise);
    }

    void display() {
        cout << "Rs. " << rupees << "." << paise << endl;
    }
};

int main() {
    Money m1(100, 75);
    Money m2(50, 50);
    Money m3;

    cout << "Money 1: ";
    m1.display();
    cout << "Money 2: ";
    m2.display();

    m3 = m1 + m2;

    cout << "Total: ";
    m3.display();

    return 0;
}

Question 7: Create a class Rectangle to represent rectangles. Overload the == operator to check if two rectangles are equal.

#include <iostream>
using namespace std;

class Rectangle {
private:
    int length, breadth;

public:
    Rectangle(int l = 0, int b = 0) : length(l), breadth(b) {}

    bool operator==(Rectangle r) {
        return (length == r.length && breadth == r.breadth);
    }

    void display() {
        cout << "Length: " << length << ", Breadth: " << breadth << endl;
    }
};

int main() {
    Rectangle r1(10, 5);
    Rectangle r2(10, 5);
    Rectangle r3(8, 4);

    cout << "Rectangle 1: ";
    r1.display();
    cout << "Rectangle 2: ";
    r2.display();
    cout << "Rectangle 3: ";
    r3.display();

    if (r1 == r2)
        cout << "Rectangle 1 and Rectangle 2 are equal" << endl;
    else
        cout << "Rectangle 1 and Rectangle 2 are not equal" << endl;

    if (r1 == r3)
        cout << "Rectangle 1 and Rectangle 3 are equal" << endl;
    else
        cout << "Rectangle 1 and Rectangle 3 are not equal" << endl;

    return 0;
}

Question 8: Define a class Time to represent time. Overload the << operator to display the time in HH:MM format.

#include <iostream>
using namespace std;

class Time {
private:
    int hours, minutes;

public:
    Time(int h = 0, int m = 0) : hours(h), minutes(m) {
        if (minutes >= 60) {
            hours += minutes / 60;
            minutes = minutes % 60;
        }
    }

    friend ostream& operator<<(ostream& out, Time& t);
};

ostream& operator<<(ostream& out, Time& t) {
    out << t.hours << ":" << t.minutes << endl;
    return out;
}

int main() {
    Time t1(9, 30);
    Time t2(14, 45);

    cout << "Time 1: " << t1;
    cout << "Time 2: " << t2;

    return 0;
}
