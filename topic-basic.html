    return true;
}

}

Question 22: Write a C++ program to check if a positive number is a palindrome or not

    int n, num, digit, rev = 0;
    cout << "Enter a positive number: ";
    cin >> num;
    n = num;
    while (num != 0) {
        digit = num % 10;
        rev = (rev * 10) + digit;
        num = num / 10;
    }
    if (n == rev) cout << n << " is a palindrome.";
    else cout << n << " is not a palindrome.";

}

Question 23: Write a C++ program to add two numbers without using any arithmetic operators

#include <iostream>
using namespace std;
int add(int a, int b) {
    while (b != 0) {
        unsigned carry = a & b;
        a = a ^ b;
        b = carry << 1;
    }
    return a;
}
int main() {
    cout << "Sum of 15 and 20: " << add(15, 20) << endl;
    return 0;
}

Question 111: Write a C++ program to convert Roman number to an integer number

#include <iostream>
#include <string>
using namespace std;

int value(char r) {
    if (r == 'I') return 1;
    if (r == 'V') return 5;
    if (r == 'X') return 10;
    if (r == 'L') return 50;
    if (r == 'C') return 100;
    if (r == 'D') return 500;
    if (r == 'M') return 1000;
    return -1;
}

int main() {
    string roman = "MCMIV";
    int res = 0;
    for (int i = 0; i < roman.length(); i++) {
        int s1 = value(roman[i]);
        if (i + 1 < roman.length()) {
            int s2 = value(roman[i + 1]);
            if (s1 >= s2) res = res + s1;
            else {
                res = res + s2 - s1;
                i++;
            }
        } else {
            res = res + s1;
        }
    }
    cout << "Integer equivalent: " << res << endl;
    return 0;
}
