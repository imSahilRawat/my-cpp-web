Question 1: Write a C++ program to create a class called Animal with a method called makeSound(). Create a derived class called Cat that overrides the makeSound() method to bark.

Question 3: Write a C++ program to create a class called Shape with a method called getArea(). Create a derived class called Rectangle that overrides the getArea() method to calculate the area of a rectangle.

#include <iostream>
using namespace std;

class Shape {
public:
    virtual float getArea() {
        return 0;
    }
};

class Rectangle : public Shape {
private:
    float length, width;
public:
    Rectangle(float l, float w) {
        length = l;
        width = w;

};

    cout << "Rectangle Area: " << rect.getArea() << endl;

    return 0;
}

Question 5: Write a C++ program to create a class known as "BankAccount" with methods called deposit() and withdraw(). Create a derived class called SavingsAccount that overrides the withdraw() method to prevent withdrawals if the account balance falls below one hundred.

#include <iostream>
using namespace std;

class BankAccount {
protected:
    float balance;
public:
    BankAccount(float initial = 0) {
        balance = initial;
    }

    void deposit(float amount) {
        balance += amount;
        cout << "Deposited Rs. " << amount << endl;
    }

    void withdraw(float amount) {
        if (amount <= balance) {
            balance -= amount;
            cout << "Withdrew Rs. " << amount << endl;
        } else {
            cout << "Insufficient balance!" << endl;
        }
    }

    void displayBalance() {
        cout << "Balance: Rs. " << balance << endl;
    }
};

class SavingsAccount : public BankAccount {
public:
    SavingsAccount(float initial = 0) : BankAccount(initial) {}

    void withdraw(float amount) {
        if (balance - amount < 100) {
            cout << "Cannot withdraw! Balance must not fall below Rs. 100" << endl;
        } else {
            balance -= amount;
            cout << "Withdrew Rs. " << amount << endl;
        }
    }
};

int main() {
    SavingsAccount acc(500);

    acc.displayBalance();
    acc.withdraw(300);
    acc.displayBalance();

    acc.withdraw(200);
    acc.displayBalance();

    return 0;
}

Question 7: Write a C++ program to create a class known as Person with methods called getFirstName() and getLastName(). Create a derived class called Employee that adds a new method named getEmployeeId() and overrides the getLastName() method to include the employee's job title.

#include <iostream>
using namespace std;

protected:
    string firstName;
    string lastName;

public:
    Person(string fn, string ln) {
        firstName = fn;
        lastName = ln;
    }

    string getFirstName() {
        return firstName;
    }

    virtual string getLastName() {
        return lastName;
    }
};

class Employee : public Person {
private:
    int employeeId;
    string jobTitle;

public:
    Employee(string fn, string ln, int id, string jt) : Person(fn, ln) {
        employeeId = id;
        jobTitle = jt;
    }

    int getEmployeeId() {
        return employeeId;
    }

    string getLastName() {
        return lastName + " (" + jobTitle + ")";
    }
};

    cout << "First Name: " << emp.getFirstName() << endl;
    cout << "Last Name: " << emp.getLastName() << endl;
    cout << "Employee ID: " << emp.getEmployeeId() << endl;

    return 0;
}

Question 8: Write a C++ program to create a class called Shape with methods called getPerimeter() and getArea(). Create a derived class called Circle that overrides the getPerimeter() and getArea() methods to calculate the area and perimeter of a circle.

#include <iostream>
using namespace std;

class Shape {
public:
    virtual float getPerimeter() {
        return 0;
    }

    virtual float getArea() {
        return 0;
    }
};

class Circle : public Shape {
private:
    float radius;
public:
    Circle(float r) {
        radius = r;
    }

    float getPerimeter() {
        return 2 * 3.14159 * radius;
    }

    float getArea() {
        return 3.14159 * radius * radius;
    }
};

int main() {
    Circle c(7);

    cout << "Circle Perimeter: " << c.getPerimeter() << endl;
    cout << "Circle Area: " << c.getArea() << endl;

    return 0;
}

Question 9: Write a C++ program to create a vehicle class hierarchy. The base class should be Vehicle, with derived classes Truck, Car and Motorcycle. Each derived class should have properties such as make, model, year, and fuel type. Implement methods for calculating fuel efficiency, distance traveled, and maximum speed.

#include <iostream>
using namespace std;

class Vehicle {
protected:
    string make;
    string model;
    int year;
    string fuelType;

public:
    Vehicle(string m, string md, int y, string f) {
        make = m;
        model = md;
        year = y;
        fuelType = f;
    }

    virtual float fuelEfficiency() = 0;
    virtual float distanceTraveled(float hours) = 0;
    virtual float maxSpeed() = 0;

    void display() {
        cout << make << " " << model << " (" << year << ") - " << fuelType << endl;
    }
};

class Car : public Vehicle {
private:
    float efficiency;
    float speed;
public:
    Car(string m, string md, int y, string f, float eff, float spd)
        : Vehicle(m, md, y, f), efficiency(eff), speed(spd) {}

    float fuelEfficiency() {
        return efficiency;
    }

    float distanceTraveled(float hours) {
        return speed * hours;
    }

    float maxSpeed() {
        return speed;
    }
};

class Truck : public Vehicle {
private:
    float efficiency;
    float speed;
    float capacity;
public:
    Truck(string m, string md, int y, string f, float eff, float spd, float cap)
        : Vehicle(m, md, y, f), efficiency(eff), speed(spd), capacity(cap) {}

    float fuelEfficiency() {
        return efficiency;
    }

    float distanceTraveled(float hours) {
        return speed * hours;
    }

    float maxSpeed() {
        return speed;
    }
};

class Motorcycle : public Vehicle {
private:
    float efficiency;
    float speed;
public:
    Motorcycle(string m, string md, int y, string f, float eff, float spd)
        : Vehicle(m, md, y, f), efficiency(eff), speed(spd) {}

    float fuelEfficiency() {
        return efficiency;
    }

    float distanceTraveled(float hours) {
        return speed * hours;
    }

    float maxSpeed() {
        return speed;
    }
};

int main() {
    Car car("Toyota", "Camry", 2024, "Petrol", 15.5, 180);
    Truck truck("Volvo", "FH16", 2023, "Diesel", 8.2, 120, 40);
    Motorcycle bike("Harley", "Iron 883", 2024, "Petrol", 22.0, 160);

    cout << "=== Car ===" << endl;
    car.display();
    cout << "Fuel Efficiency: " << car.fuelEfficiency() << " km/l" << endl;
    cout << "Distance in 2 hours: " << car.distanceTraveled(2) << " km" << endl;
    cout << "Max Speed: " << car.maxSpeed() << " km/h" << endl;

    cout << "\n=== Truck ===" << endl;
    truck.display();
    cout << "Fuel Efficiency: " << truck.fuelEfficiency() << " km/l" << endl;
    cout << "Distance in 2 hours: " << truck.distanceTraveled(2) << " km" << endl;
    cout << "Max Speed: " << truck.maxSpeed() << " km/h" << endl;

    cout << "\n=== Motorcycle ===" << endl;
    bike.display();
    cout << "Fuel Efficiency: " << bike.fuelEfficiency() << " km/l" << endl;
    cout << "Distance in 2 hours: " << bike.distanceTraveled(2) << " km" << endl;
    cout << "Max Speed: " << bike.maxSpeed() << " km/h" << endl;

    return 0;
}

Question 10: Write a C++ program that creates a class hierarchy for employees of a company. The base class should be Employee, with derived classes Manager, Developer, and Programmer. Each derived class should have properties such as name, address, salary, and job title. Implement methods for calculating bonuses, generating performance reports, and managing projects.

#include <iostream>
using namespace std;

class Employee {
protected:
    string name;
    string address;
    float salary;
    string jobTitle;

public:
    Employee(string n, string a, float s, string jt) {
        name = n;
        address = a;
        salary = s;
        jobTitle = jt;
    }

    virtual float calculateBonus() = 0;
    virtual void performanceReport() = 0;
    virtual void manageProjects() = 0;

    void display() {
        cout << "Name: " << name << ", Title: " << jobTitle << endl;
        cout << "Address: " << address << ", Salary: Rs. " << salary << endl;
    }
};

class Manager : public Employee {
private:
    int projectsManaged;
public:
    Manager(string n, string a, float s, string jt, int p)
        : Employee(n, a, s, jt), projectsManaged(p) {}

    float calculateBonus() {
        return salary * 0.20;
    }

    void performanceReport() {
        cout << "Manager " << name << " has managed " << projectsManaged << " projects successfully." << endl;
    }

    void manageProjects() {
        cout << "Manager is overseeing " << projectsManaged << " active projects." << endl;
    }
};

class Developer : public Employee {
private:
    int projectsCompleted;
    string programmingLang;
public:
    Developer(string n, string a, float s, string jt, int p, string pl)
        : Employee(n, a, s, jt), projectsCompleted(p), programmingLang(pl) {}

    float calculateBonus() {
        return salary * 0.15;
    }

    void performanceReport() {
        cout << "Developer " << name << " has completed " << projectsCompleted << " projects." << endl;
    }

    void manageProjects() {
        cout << "Developer is working on projects using " << programmingLang << "." << endl;
    }
};

class Programmer : public Employee {
private:
    int linesOfCode;
    string primaryLang;
public:
    Programmer(string n, string a, float s, string jt, int loc, string pl)
        : Employee(n, a, s, jt), linesOfCode(loc), primaryLang(pl) {}

    float calculateBonus() {
        return salary * 0.10;
    }

    void performanceReport() {
        cout << "Programmer " << name << " has written " << linesOfCode << " lines of code." << endl;
    }

    void manageProjects() {
        cout << "Programmer is coding in " << primaryLang << "." << endl;
    }
};

int main() {
    Manager mgr("Alice Johnson", "123 Main St", 80000, "Senior Manager", 5);
    Developer dev("Bob Smith", "456 Oak Ave", 60000, "Lead Developer", 12, "Java");
    Programmer pgm("Carol White", "789 Pine Rd", 45000, "Junior Programmer", 5000, "Python");

    cout << "=== Manager ===" << endl;
    mgr.display();
    cout << "Bonus: Rs. " << mgr.calculateBonus() << endl;
    mgr.performanceReport();
    mgr.manageProjects();

    cout << "\n=== Developer ===" << endl;
    dev.display();
    cout << "Bonus: Rs. " << dev.calculateBonus() << endl;
    dev.performanceReport();
    dev.manageProjects();

    cout << "\n=== Programmer ===" << endl;
    pgm.display();
    cout << "Bonus: Rs. " << pgm.calculateBonus() << endl;
    pgm.performanceReport();
    pgm.manageProjects();

    return 0;
}
