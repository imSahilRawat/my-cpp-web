Question 1: Program to Check Anagram

#include <iostream>
#include <string>
using namespace std;

class StringAnalyzer {
private:
    string str1, str2;

public:
    StringAnalyzer(string s1, string s2) {
        str1 = s1;
        str2 = s2;
    }

    bool areAnagram() {
        string temp1 = str1, temp2 = str2;

        // Remove spaces
        temp1.erase(remove(temp1.begin(), temp1.end(), ' '), temp1.end());
        temp2.erase(remove(temp2.begin(), temp2.end(), ' '), temp2.end());

        // Convert to lowercase
        transform(temp1.begin(), temp1.end(), temp1.begin(), ::tolower);
        transform(temp2.begin(), temp2.end(), temp2.begin(), ::tolower);

        // Sort and compare
        sort(temp1.begin(), temp1.end());
        sort(temp2.begin(), temp2.end());

        return temp1 == temp2;
    }

    void display() {
        cout << "String 1: " << str1 << endl;
        cout << "String 2: " << str2 << endl;
    }
};

int main() {
    string s1, s2;

    cout << "Enter first string: ";
    getline(cin, s1);

    cout << "Enter second string: ";
    getline(cin, s2);

    StringAnalyzer analyzer(s1, s2);

    analyzer.display();

    if (analyzer.areAnagram()) {
        cout << "The strings are ANAGRAMS!" << endl;
    } else {
        cout << "The strings are NOT anagrams!" << endl;
    }

    return 0;
}

class StringDivider {
private:
    string str;
    int n;

public:
    StringDivider(string s, int num) {
        str = s;
        n = num;
    }

    void divideString() {
        int len = str.length();
        int partSize = len / n;
        int remainder = len % n;

        if (len < n) {
            cout << "String length is less than N parts!" << endl;
            return;
        }

        int start = 0;
        for (int i = 0; i < n; i++) {
            int size = partSize + (i < remainder ? 1 : 0);
            cout << "Part " << (i + 1) << ": " << str.substr(start, size) << endl;
            start += size;
        }
    }
};

int main() {
    string input;
    int n;

    cout << "Enter a string: ";
    getline(cin, input);

    cout << "Enter number of parts (N): ";
    cin >> n;

    StringDivider divider(input, n);
    divider.divideString();

    return 0;
}

        cout << "All permutations of the string:" << endl;
        findPermutations(str, 0, str.length() - 1);

    void findPermutations(string s, int left, int right) {
        if (left == right) {
            cout << s << endl;

        }
    }
};

class LongestRepeatingSequence {
private:
    string str;

public:
    LongestRepeatingSequence(string s) {
        str = s;
    }

    void findLongestRepeating() {
        int n = str.length();
        string longest = "";

        for (int len = 1; len <= n / 2; len++) {
            for (int i = 0; i <= n - 2 * len; i++) {
                string sub1 = str.substr(i, len);
                for (int j = i + len; j <= n - len; j++) {
                    string sub2 = str.substr(j, len);
                    if (sub1 == sub2 && len > (int)longest.length()) {
                        longest = sub1;
                    }
                }
            }
        }

        if (longest != "") {
            cout << "Longest repeating sequence: " << longest << endl;
        } else {
            cout << "No repeating sequence found!" << endl;
        }
    }
};

        int left = 0;
        int right = s.length() - 1;

        while (left < right) {
            if (s[left] != s[right]) {
                return false;
            }
            left++;
            right--;
        }
        return true;
    }

    void findSmallestBiggestPalindrome() {
        string smallest = "", biggest = "";
        int smallestLen = 9999, biggestLen = 0;
        int wordStart = 0;

                        if (len < smallestLen) {
                            smallestLen = len;
                            smallest = word;
                        }
                        if (len > biggestLen) {
                            biggestLen = len;
                            biggest = word;
                        }
                    }
                }
                wordStart = i + 1;
            }
        }

        if (smallest != "") {
            cout << "Smallest palindrome word: " << smallest << endl;

        }
    }

class StringSplitter {
private:
    string str;
    int n;

public:
    StringSplitter(string s, int num) {
        str = s;
        n = num;
    }

    void splitString() {
        int len = str.length();
        int partSize = len / n;
        int remainder = len % n;

        if (len < n) {
            cout << "String length is less than N parts!" << endl;
            return;
        }

        int start = 0;
        for (int i = 0; i < n; i++) {
            int size = partSize + (i < remainder ? 1 : 0);
            cout << "Part " << (i + 1) << ": " << str.substr(start, size) << endl;
            start += size;
        }
    }
};

int main() {
    string input;
    int n;

    cout << "Enter a string: ";
    getline(cin, input);

    cout << "Enter number of parts (N): ";
    cin >> n;

    StringSplitter splitter(input, n);
    splitter.splitString();

    return 0;
}
