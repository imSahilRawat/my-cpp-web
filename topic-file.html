Question 6: Write a C++ program to find and replace a specific word in a text file.

#include <iostream>
#include <fstream>
#include <string>
using namespace std;

int main() {
    char filename[100], oldWord[50], newWord[50];

    cout << "Enter filename: ";
    cin >> filename;
    cout << "Enter word to replace: ";
    cin >> oldWord;
    cout << "Enter new word: ";
    cin >> newWord;

    ifstream infile(filename);
    ofstream outfile("temp.txt");

    if (!infile || !outfile) {
        cout << "Error opening files!" << endl;
        return 1;
    }

    char word[100];
    while (infile >> word) {
        int match = 1;
        int i = 0;
        while (oldWord[i] != '\0' && word[i] != '\0') {
            if (oldWord[i] != word[i]) {
                match = 0;
                break;
            }
            i++;
        }
        if (oldWord[i] == '\0' && word[i] == '\0' && match) {
            outfile << newWord << " ";
        } else {
            outfile << word << " ";
        }
    }

    infile.close();
    outfile.close();

    remove(filename);
    rename("temp.txt", filename);

    cout << "Word replaced successfully!" << endl;
    return 0;
}

Question 8: Write a C++ program to sort the lines of a text file in alphabetical order.

#include <iostream>
#include <fstream>
using namespace std;

int main() {
    ifstream infile("output.txt");

    if (!infile) {
        cout << "Error opening file!" << endl;
        return 1;
    }

    char lines[100][256];
    int lineCount = 0;

    while (infile.getline(lines[lineCount], 256) && lineCount < 100) {
        lineCount++;
    }
    infile.close();

    // Bubble sort for lines
    for (int i = 0; i < lineCount - 1; i++) {
        for (int j = 0; j < lineCount - i - 1; j++) {
            if (lines[j][0] > lines[j + 1][0]) {
                char temp[256];
                int k = 0;
                while (lines[j][k] != '\0') {
                    temp[k] = lines[j][k];
                    k++;
                }
                temp[k] = '\0';

                k = 0;
                while (lines[j + 1][k] != '\0') {
                    lines[j][k] = lines[j + 1][k];
                    k++;
                }
                lines[j][k] = '\0';

                k = 0;
                while (temp[k] != '\0') {
                    lines[j + 1][k] = temp[k];
                    k++;
                }
                lines[j + 1][k] = '\0';
            }
        }
    }

    ofstream outfile("output.txt");
    for (int i = 0; i < lineCount; i++) {
        outfile << lines[i] << endl;
    }
    outfile.close();

    cout << "Lines sorted successfully!" << endl;
    return 0;
}

Question 10: Write a C++ program to split a large text file into smaller files of equal size.

#include <iostream>
#include <fstream>
using namespace std;

int main() {
    ifstream infile("largefile.txt");

    if (!infile) {
        cout << "Error opening file!" << endl;
        return 1;
    }

    int linesPerFile, fileNum = 1;
    cout << "Enter number of lines per file: ";
    cin >> linesPerFile;

    char outFilename[50];
    ofstream outfile;
    int lineCount = 0;

    char line[256];
    while (infile.getline(line, 256)) {
        if (lineCount == 0) {
            sprintf(outFilename, "part%d.txt", fileNum);
            outfile.open(outFilename);
            fileNum++;
        }

        outfile << line << endl;
        lineCount++;

        if (lineCount == linesPerFile) {
            outfile.close();
            lineCount = 0;
        }
    }

    if (outfile.is_open()) {
        outfile.close();
    }
    infile.close();

    cout << "File split successfully!" << endl;
    return 0;
}

Question 11: Write a C++ program to search for a specific string in a text file and display its line number(s).

#include <iostream>
#include <fstream>
using namespace std;

int containsString(char str[], char search[]) {
    int i = 0, j = 0;
    while (str[i] != '\0' && search[j] != '\0') {
        if (str[i] == search[j]) {
            i++;
            j++;
        } else {
            i = i - j + 1;
            j = 0;
        }
    }
    if (search[j] == '\0')
        return 1;
    return 0;
}

int main() {
    char filename[100], searchStr[100];

    cout << "Enter filename: ";
    cin >> filename;
    cout << "Enter string to search: ";
    cin >> searchStr;

    ifstream infile(filename);

    if (!infile) {
        cout << "Error opening file!" << endl;
        return 1;
    }

    char line[256];
    int lineNum = 0, found = 0;

    while (infile.getline(line, 256)) {
        lineNum++;
        if (containsString(line, searchStr)) {
            cout << "Found at line " << lineNum << ": " << line << endl;
            found++;
        }
    }

    if (found == 0) {
        cout << "String not found!" << endl;
    }

    infile.close();
    return 0;
}

Question 12: Write a C++ program to encrypt the contents of a text file using a simple encryption algorithm.

#include <iostream>
#include <fstream>
using namespace std;

int main() {
    char filename[100];
    int key;

    cout << "Enter filename: ";
    cin >> filename;
    cout << "Enter encryption key (1-25): ";
    cin >> key;

    ifstream infile(filename);
    ofstream outfile("encrypted.txt");

    if (!infile || !outfile) {
        cout << "Error opening files!" << endl;
        return 1;
    }

    char ch;
    while (infile.get(ch)) {
        if (ch >= 'a' && ch <= 'z') {
            ch = ((ch - 'a' + key) % 26) + 'a';
        } else if (ch >= 'A' && ch <= 'Z') {
            ch = ((ch - 'A' + key) % 26) + 'A';
        }
        outfile.put(ch);
    }

    infile.close();
    outfile.close();

    cout << "File encrypted successfully!" << endl;
    return 0;
}

Question 15: Write a C++ program to calculate the average of numbers stored in a file.

#include <iostream>
#include <fstream>
using namespace std;

int main() {
    ifstream infile("numbers.txt");

    if (!infile) {
        cout << "Error opening file!" << endl;
        return 1;
    }

    int num, sum = 0, count = 0;

    while (infile >> num) {
        sum += num;
        count++;
    }

    if (count > 0) {
        double average = (double)sum / count;
        cout << "Total numbers: " << count << endl;
        cout << "Sum: " << sum << endl;
        cout << "Average: " << average << endl;
    } else {
        cout << "No numbers found in file!" << endl;
    }

    infile.close();
    return 0;
}

Question 16: Write a C++ program to store the information of 10 Students having rollno,name,marks,mobileno properties, get() and set() member functions as binary file with following options 1. Insert Record 2. Delete Record 3. Update Record (based on rollno) 4. Search Record (based on rollno) 5. Clone the file 6. Exit.

#include <iostream>
#include <fstream>
#include <cstring>
using namespace std;

class Student {
    int rollno;
    char name[50];
    float marks;
    char mobileno[15];

public:
    void get() {
        cout << "Enter Roll No: ";
        cin >> rollno;
        cout << "Enter Name: ";
        cin >> name;
        cout << "Enter Marks: ";
        cin >> marks;
        cout << "Enter Mobile No: ";
        cin >> mobileno;
    }

    void set(int r, char n[], float m, char mob[]) {
        rollno = r;
        strcpy(name, n);
        marks = m;
        strcpy(mobileno, mob);
    }

    void display() {
        cout << rollno << "\t" << name << "\t" << marks << "\t" << mobileno << endl;
    }

    int getRollno() {
        return rollno;
    }
};

void insertRecord() {
    Student s;
    s.get();
    ofstream outfile("students.dat", ios::binary | ios::app);
    outfile.write((char*)&s, sizeof(s));
    outfile.close();
    cout << "Record inserted successfully!" << endl;
}

void deleteRecord() {
    int rollno, found = 0;
    cout << "Enter Roll No to delete: ";
    cin >> rollno;

    ifstream infile("students.dat", ios::binary);
    ofstream outfile("temp.dat", ios::binary);

    Student s;
    while (infile.read((char*)&s, sizeof(s))) {
        if (s.getRollno() != rollno) {
            outfile.write((char*)&s, sizeof(s));
        } else {
            found = 1;
        }
    }

    infile.close();
    outfile.close();

    remove("students.dat");
    rename("temp.dat", "students.dat");

    if (found)
        cout << "Record deleted successfully!" << endl;
    else
        cout << "Record not found!" << endl;
}

void updateRecord() {
    int rollno, found = 0;
    cout << "Enter Roll No to update: ";
    cin >> rollno;

    fstream file("students.dat", ios::binary | ios::in | ios::out);
    Student s;

    while (file.read((char*)&s, sizeof(s))) {
        if (s.getRollno() == rollno) {
            found = 1;
            int pos = file.tellg();
            pos = pos - sizeof(s);
            file.seekg(pos);
            s.get();
            file.write((char*)&s, sizeof(s));
            break;
        }
    }

    file.close();

    if (found)
        cout << "Record updated successfully!" << endl;
    else
        cout << "Record not found!" << endl;
}

void searchRecord() {
    int rollno, found = 0;
    cout << "Enter Roll No to search: ";
    cin >> rollno;

    ifstream infile("students.dat", ios::binary);
    Student s;

    while (infile.read((char*)&s, sizeof(s))) {
        if (s.getRollno() == rollno) {
            found = 1;
            cout << "Record found:" << endl;
            s.display();
            break;
        }
    }

    infile.close();

    if (!found)
        cout << "Record not found!" << endl;
}

void cloneFile() {
    ifstream infile("students.dat", ios::binary);
    ofstream outfile("students_clone.dat", ios::binary);

    Student s;
    while (infile.read((char*)&s, sizeof(s))) {
        outfile.write((char*)&s, sizeof(s));
    }

    infile.close();
    outfile.close();

    cout << "File cloned successfully!" << endl;
}

int main() {
    int choice;

    do {
        cout << "\n===== Student Record System =====" << endl;
        cout << "1. Insert Record" << endl;
        cout << "2. Delete Record" << endl;
        cout << "3. Update Record" << endl;
        cout << "4. Search Record" << endl;
        cout << "5. Clone the file" << endl;
        cout << "6. Exit" << endl;
        cout << "Enter your choice: ";
        cin >> choice;

        switch (choice) {
            case 1:
                insertRecord();
                break;
            case 2:
                deleteRecord();
                break;
            case 3:
                updateRecord();
                break;
            case 4:
                searchRecord();
                break;
            case 5:
                cloneFile();
                break;
            case 6:
                cout << "Exiting..." << endl;
                break;
            default:
                cout << "Invalid choice!" << endl;
        }
    } while (choice != 6);

    return 0;
}
