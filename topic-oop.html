#include <iostream>
using namespace std;

float customSqrt(float n) {
    if (n <= 0) return 0;
    float x = n;
    float y = 1;
    float e = 0.000001;
    while(x - y > e) {
        x = (x + y) / 2;
        y = n / x;
    }
    return x;
}

class Shape {
public:
    virtual float area() = 0;
    virtual float perimeter() = 0;
    virtual ~Shape() {}
};

class Rectangle : public Shape {
    float width, height;
public:
    Rectangle(float w, float h) : width(w), height(h) {}

    float area() {
        return width * height;
    }

    float perimeter() {
        return 2 * (width + height);
    }
};

class Circle : public Shape {
    float radius;
public:
    Circle(float r) : radius(r) {}

    float area() {
        return 3.14159 * radius * radius;
    }

    float perimeter() {
        return 2 * 3.14159 * radius;
    }
};

class Triangle : public Shape {
    float a, b, c;
public:
    Triangle(float x, float y, float z) : a(x), b(y), c(z) {}

    float area() {
        float s = (a + b + c) / 2;
        return customSqrt(s * (s - a) * (s - b) * (s - c));
    }

    float perimeter() {
        return a + b + c;
    }
};

int main() {
    Rectangle r(10, 5);
    Circle c(7);
    Triangle t(3, 4, 5);

    cout << "Rectangle: Area = " << r.area() << ", Perimeter = " << r.perimeter() << endl;
    cout << "Circle: Area = " << c.area() << ", Perimeter = " << c.perimeter() << endl;
    cout << "Triangle: Area = " << t.area() << ", Perimeter = " << t.perimeter() << endl;

    return 0;
}

    int itemCount;
    float ratings[10];
    int ratingCount;

public:
    Restaurant() {
        itemCount = ratingCount = 0;
    }

    void addItem(string name, float price) {
        items[itemCount++] = new MenuItem(name, price);
        cout << "Item added!" << endl;
    }

            if (items[i]->name == name) {

        type = t;

        cout << type << " Booking #" << id << ": " << details << endl;
    }

class TravelService {
    Reservation* reservations[100];
    int count;
    int nextId;

public:
    TravelService() {
        count = 0;
        nextId = 1;
    }

    Reservation* searchFlight(string destination) {
        cout << "Searching flights to " << destination << "..." << endl;
        cout << "Flight found: AI101 to " << destination << " at 10:00 AM" << endl;
        return NULL;
    }

        return NULL;

        string details = destination + " on " + date;
        reservations[count++] = new Reservation(nextId++, "Flight", details);
        cout << "Flight booked successfully! Confirmation #" << (nextId - 1) << endl;
        return nextId - 1;
    }

    int bookHotel(string city, string date) {
        string details = city + " from " + date;
        reservations[count++] = new Reservation(nextId++, "Hotel", details);
        cout << "Hotel booked successfully! Confirmation #" << (nextId - 1) << endl;
        return nextId - 1;

            if (reservations[i]->id == id && reservations[i]->active) {
                reservations[i]->active = false;
                cout << "Reservation #" << id << " cancelled!" << endl;
                return;
            }
        }
        cout << "Reservation not found!" << endl;
    }

    void displayReservations() {
        cout << "\n=== Your Reservations ===" << endl;
        for (int i = 0; i < count; i++) {
            if (reservations[i]->active) {
                reservations[i]->display();
            }
        }
    }
};
