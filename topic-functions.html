Question 1: Write a C++ method to find the smallest number among three numbers.

#include <iostream>
using namespace std;
int findSmallest(int a, int b, int c) {
    int smallest = a;
    if (b < smallest) {
        smallest = b;
    }
    if (c < smallest) {
        smallest = c;
    }
    return smallest;
}
int main() {
    int x, y, z;    
    cout << "Enter three numbers: ";
    cin >> x >> y >> z;
    cout << "Smallest number is: " << findSmallest(x, y, z);
 return 0;
}

Question 4: Write a C++ method to find all twin prime numbers less than 100.

#include <iostream>
using namespace std;
int isPrime(int n) {
    if (n < 2)
        return 0;
    for (int i = 2; i <= n / 2; i++) {
        if (n % i == 0)
            return 0;
    }
    return 1;
}
void findTwinPrimes() {
    for (int i = 2; i < 100; i++) {
        if (isPrime(i) && isPrime(i + 2)) {
            cout << "(" << i << ", " << i + 2 << ")" << endl;
        }
    }
}
int main() {
    cout << "Twin prime numbers less than 100 are:\n";
    findTwinPrimes();
    return 0;
}

Question 5: Write a C++ method to count the number of digits in an integer with the value 2. The integer may be assumed to be non-negative.

#include <iostream>
using namespace std;
int countTwos(int num) {
    int count = 0;
    while (num > 0) {
        if (num % 10 == 2) {
            count++;
        }
        num = num / 10;
    }
    return count;
}
int main() {
    int n;
    cout << "Enter a non-negative integer: ";
    cin >> n;
    cout << "Number of digits with value 2: " << countTwos(n);
    return 0;
}

Question 6: Write a C++ method that accepts three integers and checks whether they are consecutive or not. Returns true or false.

#include <iostream>
using namespace std;
int areConsecutive(int a, int b, int c) {
    int minVal = a;
    if (b < minVal) minVal = b;
    if (c < minVal) minVal = c;
    int maxVal = a;
    if (b > maxVal) maxVal = b;
    if (c > maxVal) maxVal = c;
    int midVal = (a + b + c) - (minVal + maxVal);
    if (midVal == minVal + 1 && maxVal == midVal + 1)
        return 1;
    else
        return 0;
}
int main() {
    int x, y, z;
    cout << "Enter three integers: ";
    cin >> x >> y >> z;
    if (areConsecutive(x, y, z) == 1)
        cout << "Numbers are consecutive";
    else
        cout << "Numbers are NOT consecutive";
    return 0;
}

Question 7: Write a C++ method that checks whether all the characters in a given string are vowels (a,e,i,o,u) or not. Return true if each character in the string is a vowel, otherwise return false.

#include <iostream>
#include <string>
using namespace std;
int allVowels(string str) {
    for (int i = 0; i < str.length(); i++) {
        char ch = str[i];

        if (!(ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u' || ch == 'A' || ch == 'E' || ch == 'I' || ch == 'O' || ch == 'U')) {
            return 0;   
        }
    }
    return 1;   
}
int main() {
    string text;
    cout << "Enter a string: ";
    cin >> text;
    if (allVowels(text) == 1)
        cout << "All characters are vowels";
    else
        cout << "Not all characters are vowels";
    return 0;
}
