Question 3: Write a C++ program to create an abstract class Shape with abstract methods calculateArea() and calculatePerimeter(). Create derived classes Circle and Triangle that extend the Shape class and implement the respective methods to calculate the area and perimeter of each shape.

#include <iostream>
using namespace std;

class Shape {
public:
    virtual float calculateArea() = 0;
    virtual float calculatePerimeter() = 0;
    virtual ~Shape() {}
};

class Circle : public Shape {
private:
    float radius;
public:
    Circle(float r) : radius(r) {}

    float calculateArea() {
        return 3.14159 * radius * radius;
    }

    float calculatePerimeter() {
        return 2 * 3.14159 * radius;
    }
};

class Triangle : public Shape {
private:
    float a, b, c;
public:
    Triangle(float x, float y, float z) : a(x), b(y), c(z) {}

    float calculateArea() {
        float s = (a + b + c) / 2;
        return s * (s - a) * (s - b) * (s - c);
    }

    float calculatePerimeter() {
        return a + b + c;
    }
};

int main() {
    Circle c(5);
    Triangle t(3, 4, 5);

    cout << "Circle:" << endl;
    cout << "Area: " << c.calculateArea() << endl;
    cout << "Perimeter: " << c.calculatePerimeter() << endl;

    cout << "\nTriangle:" << endl;
    cout << "Area: " << t.calculateArea() << endl;
    cout << "Perimeter: " << t.calculatePerimeter() << endl;

    return 0;
}

Question 4: Write a C++ program to create an abstract class BankAccount with abstract methods deposit() and withdraw(). Create derived classes: SavingsAccount and CurrentAccount that extend the BankAccount class and implement the respective methods to handle deposits and withdrawals for each account type.

#include <iostream>
using namespace std;

class BankAccount {
protected:
    float balance;
public:
    BankAccount(float initial = 0) : balance(initial) {}

    virtual void deposit(float amount) = 0;
    virtual void withdraw(float amount) = 0;

    void displayBalance() {
        cout << "Balance: Rs. " << balance << endl;
    }

    virtual ~BankAccount() {}
};

class SavingsAccount : public BankAccount {
public:
    SavingsAccount(float initial = 0) : BankAccount(initial) {}

    void deposit(float amount) {
        balance += amount;
        cout << "Deposited Rs. " << amount << " in Savings Account" << endl;
    }

    void withdraw(float amount) {
        if (amount <= balance) {
            balance -= amount;
            cout << "Withdrew Rs. " << amount << " from Savings Account" << endl;
        } else {
            cout << "Insufficient balance!" << endl;
        }
    }
};

class CurrentAccount : public BankAccount {
private:
    float overdraftLimit;
public:
    CurrentAccount(float initial = 0, float limit = 1000) : BankAccount(initial), overdraftLimit(limit) {}

    void deposit(float amount) {
        balance += amount;
        cout << "Deposited Rs. " << amount << " in Current Account" << endl;
    }

    void withdraw(float amount) {
        if (amount <= balance + overdraftLimit) {
            balance -= amount;
            cout << "Withdrew Rs. " << amount << " from Current Account" << endl;
        } else {
            cout << "Exceeds overdraft limit!" << endl;
        }
    }
};

int main() {
    SavingsAccount savings(5000);
    CurrentAccount current(10000, 2000);

    cout << "=== Savings Account ===" << endl;
    savings.deposit(2000);
    savings.withdraw(1000);
    savings.displayBalance();

    cout << "\n=== Current Account ===" << endl;
    current.deposit(5000);
    current.withdraw(3000);
    current.displayBalance();

    return 0;
}

Question 6: Write a C++ program to create an abstract class Employee with abstract methods calculateSalary() and displayInfo(). Create derived classes Manager and Programmer that extend the Employee class and implement the respective methods to calculate salary and display information for each role.

#include <iostream>
using namespace std;

class Employee {
protected:
    string name;
    int id;
    float baseSalary;
public:
    Employee(int i, string n, float s) {
        id = i;
        name = n;
        baseSalary = s;
    }

    virtual void calculateSalary() = 0;
    virtual void displayInfo() = 0;
    virtual ~Employee() {}
};

class Manager : public Employee {
private:
    float bonus;
public:
    Manager(int i, string n, float s, float b) : Employee(i, n, s), bonus(b) {}

    void calculateSalary() {
        float total = baseSalary + bonus;
        cout << "Manager's total salary: Rs. " << total << endl;
    }

    void displayInfo() {
        cout << "Manager - ID: " << id << ", Name: " << name << ", Base: Rs. " << baseSalary << ", Bonus: Rs. " << bonus << endl;
    }
};

class Programmer : public Employee {
private:
    int hoursWorked;
    float hourlyRate;
public:
    Programmer(int i, string n, float s, int h, float r) : Employee(i, n, s), hoursWorked(h), hourlyRate(r) {}

    void calculateSalary() {
        float overtime = hoursWorked * hourlyRate;
        float total = baseSalary + overtime;
        cout << "Programmer's total salary: Rs. " << total << endl;
    }

    void displayInfo() {
        cout << "Programmer - ID: " << id << ", Name: " << name << ", Base: Rs. " << baseSalary << ", Overtime Hours: " << hoursWorked << endl;
    }
};

int main() {
    Manager mgr(1, "John", 50000, 10000);
    Programmer pgmr(2, "Alice", 40000, 20, 500);

    cout << "=== Manager ===" << endl;
    mgr.displayInfo();
    mgr.calculateSalary();

    cout << "\n=== Programmer ===" << endl;
    pgmr.displayInfo();
    pgmr.calculateSalary();

    return 0;
}

Question 7: Write a C++ program to create an abstract class Shape3D with abstract methods calculateVolume() and calculateSurfaceArea(). Create derived classes Sphere and Cube that extend the Shape3D class and implement the respective methods to calculate the volume and surface area of each shape.

#include <iostream>
using namespace std;

class Shape3D {
public:
    virtual float calculateVolume() = 0;
    virtual float calculateSurfaceArea() = 0;
    virtual ~Shape3D() {}
};

class Sphere : public Shape3D {
private:
    float radius;
public:
    Sphere(float r) : radius(r) {}

    float calculateVolume() {
        return (4.0 / 3.0) * 3.14159 * radius * radius * radius;
    }

    float calculateSurfaceArea() {
        return 4 * 3.14159 * radius * radius;
    }
};

class Cube : public Shape3D {
private:
    float side;
public:
    Cube(float s) : side(s) {}

    float calculateVolume() {
        return side * side * side;
    }

    float calculateSurfaceArea() {
        return 6 * side * side;
    }
};

int main() {
    Sphere s(5);
    Cube c(4);

    cout << "=== Sphere (radius = 5) ===" << endl;
    cout << "Volume: " << s.calculateVolume() << endl;
    cout << "Surface Area: " << s.calculateSurfaceArea() << endl;

    cout << "\n=== Cube (side = 4) ===" << endl;
    cout << "Volume: " << c.calculateVolume() << endl;
    cout << "Surface Area: " << c.calculateSurfaceArea() << endl;

    return 0;
}

Question 11: Write a C++ program to create an abstract class Shape2D with abstract methods draw() and resize(). Create derived classes Rectangle and Circle that extend the Shape2D class and implement the respective methods to draw and resize each shape.

#include <iostream>
using namespace std;

class Shape2D {
protected:
    float size;
public:
    Shape2D(float s = 0) : size(s) {}
    virtual void draw() = 0;
    virtual void resize(float newSize) = 0;
    virtual ~Shape2D() {}
};

class Rectangle : public Shape2D {
private:
    float width, height;
public:
    Rectangle(float w, float h) : width(w), height(h) {}

    void draw() {
        cout << "Rectangle: Drawing rectangle with width " << width << " and height " << height << endl;
    }

    void resize(float factor) {
        width *= factor;
        height *= factor;
        cout << "Rectangle: Resized to width " << width << " and height " << height << endl;
    }
};

class Circle : public Shape2D {
private:
    float radius;
public:
    Circle(float r) : radius(r) {}

    void draw() {
        cout << "Circle: Drawing circle with radius " << radius << endl;
    }

    void resize(float factor) {
        radius *= factor;
        cout << "Circle: Resized to radius " << radius << endl;
    }
};

int main() {
    Rectangle rect(10, 5);
    Circle circle(7);

    cout << "=== Rectangle ===" << endl;
    rect.draw();
    rect.resize(1.5);

    cout << "\n=== Circle ===" << endl;
    circle.draw();
    circle.resize(2);

    return 0;
}

Question 12: Write a C++ program to create an abstract class GeometricShape with abstract methods area() and perimeter(). Create derived classes Triangle and Square that extend the GeometricShape class and implement the respective methods to calculate the area and perimeter of each shape.

#include <iostream>
using namespace std;

class GeometricShape {
public:
    virtual float area() = 0;
    virtual float perimeter() = 0;
    virtual ~GeometricShape() {}
};

class Triangle : public GeometricShape {
private:
    float a, b, c;
public:
    Triangle(float x, float y, float z) : a(x), b(y), c(z) {}

    float area() {
        float s = (a + b + c) / 2;
        return s * (s - a) * (s - b) * (s - c);
    }

    float perimeter() {
        return a + b + c;
    }
};

class Square : public GeometricShape {
private:
    float side;
public:
    Square(float s) : side(s) {}

    float area() {
        return side * side;
    }

    float perimeter() {
        return 4 * side;
    }
};

int main() {
    Triangle t(3, 4, 5);
    Square s(6);

    cout << "=== Triangle (3, 4, 5) ===" << endl;
    cout << "Area: " << t.area() << endl;
    cout << "Perimeter: " << t.perimeter() << endl;

    cout << "\n=== Square (side = 6) ===" << endl;
    cout << "Area: " << s.area() << endl;
    cout << "Perimeter: " << s.perimeter() << endl;

    return 0;
}
